The **IWithdrawalQueue** manages the liquidity exit process for Syndicates. Because capital may be locked in active policies or external strategies, withdrawals are not always instant. This contract handles the queueing, maturation, and execution of withdrawal requests.

## Interface

```solidity
// SPDX-License-Identifier: BUSL-1.1
pragma solidity ^0.8.20;

import {Types} from "../libraries/helpers/Types.sol";

interface IWithdrawalQueue {
    /* ========= Admin ========= */

    function setCapitalPool(address capitalPool) external;

    /* ========= Mutations ========= */

    function enqueue(address user, uint256 shares, uint256 unlockTimestamp) external returns (uint256 idx);
    function cancel(address user, uint256 requestIndex) external returns (uint256 sharesCancelled);
    function reduceRequest(address user, uint256 requestIndex, uint256 sharesToReduce) external;
    function pop(address user, uint256 requestIndex) external;

    /* ========= Views ========= */

    function requestCount(address user) external view returns (uint256);
    function totalPendingShares(address user) external view returns (uint256);
    function getRequest(address user, uint256 requestIndex) external view returns (Types.WithdrawalRequest memory);
    function listRequests(address user) external view returns (uint256[] memory shares, uint256[] memory unlocks);
    function pagedRequests(address user, uint256 offset, uint256 limit)
        external
        view
        returns (uint256[] memory shares, uint256[] memory unlocks);
    function maturedRequests(address user, uint256 currentTimestamp)
        external
        view
        returns (uint256[] memory shares, uint256[] memory unlocks);
    function maturedShares(address user, uint256 currentTimestamp) external view returns (uint256);
    function nextUnlock(address user, uint256 currentTimestamp)
        external
        view
        returns (bool exists, uint256 unlockTimestamp, uint256 shares);
}
```
